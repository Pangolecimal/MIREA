```js
let x = [0, 86, 87],
    [a, b, c, y0] = [3, 2, 4, 4]

function y(i) {
    if (i === 0) return y0
    let sum = 0

    for (let j = 0; j <= a; j++) {
        for (let k = 0; k <= b; k++) {
            sum += y(i - 1) ** j * x[i] ** k
        }
    }

    return sum % c
}

console.log(...x.map((_, i) => y(i))) // 4 3 0
```
